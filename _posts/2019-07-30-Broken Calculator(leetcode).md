class Solution(object):
    def brokenCalc(self, X, Y):
        if X > Y:
            return X - Y
        elif X == Y:
            return 0
        count = 0
        while X < Y:
            if Y % 2 == 0:
                Y = Y / 2
                count += 1
            else:
                Y = (Y + 1) / 2
                count += 2
        count += X - Y
        return count
